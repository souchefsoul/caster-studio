  id: string

  isActive: boolean
  createdAt: string
}

2. Create `src/lib/brandModels.ts` with these exported functions:

   - `fetchBrandModels(userId: string): Promise<BrandModel[]>` — Fetches all brand models for user from `brand_models` table, ordered by `created_at DESC`. Maps DB columns: `id`, `name`, `face_image_url` -> `faceImageUrl`, `is_active` -> `isActive`, `created_at` -> `createdAt`.

   - `createBrandModel(userId: string, name: string, faceImageUrl: string): Promise<BrandModel>` — Inserts a new row into `brand_models` with `user_id`, `name`, `face_image_url`, `is_active: false`. Returns the created model mapped to frontend type. Uses `supabase.from('brand_models').insert(...).select().single()`.

   - `setActiveBrandModel(userId: string, modelId: string): Promise<void>` — First sets ALL user's brand models to `is_active: false` via `supabase.from('brand_models').update({ is_active: false }).eq('user_id', userId)`, then sets the target model to `is_active: true` via `supabase.from('brand_models').update({ is_active: true }).eq('id', modelId)`.

   - `deleteBrandModel(modelId: string): Promise<void>` — Deletes the brand model row: `supabase.from('brand_models').delete().eq('id', modelId)`.

3. Create `src/hooks/useBrandModels.ts`:

   - Export `useBrandModels()` hook with React state:
     - `models: BrandModel[]` — list of user's brand models
     - `loading: boolean` — true during initial fetch
     - `activeModel: BrandModel | null` — derived: `models.find(m => m.isActive) ?? null`
   - On mount (when user exists from `useAuth()`), calls `fetchBrandModels(user.id)` and sets models state.
   - Exposes methods that update both local state and DB:
     - `create(name: string, faceImageUrl: string)` — calls `createBrandModel`, adds result to models state
     - `remove(modelId: string)` — calls `deleteBrandModel`, removes from models state
     - `setActive(modelId: string)` — calls `setActiveBrandModel`, updates isActive in local models state
   - Returns `{ models, activeModel, loading, create, remove, setActive }`

4. Add to workspace store in `src/stores/workspaceStore.ts`:
   - Add `activeBrandFaceUrl: string | null` to interface and initial state (null)
   - Add `setActiveBrandFaceUrl: (url: string | null) => void` action
   - This is the URL that generation modes can read to include the brand face in their requests
